Section 1

Variables and Types

Variables in Rust

Variables are immutable by default!


fn main() {
    let x = 5;
    println!("x = {}", x);

    // x = 6;  // ERROR: cannot assign twice to immutable variable
}

Why?

  • Prevents accidental mutations
  • Enables compiler optimizations

Mutable Variables

Use mut to make a variable mutable:


fn main() {
    let mut x = 5;
    println!("x = {}", x);

    x = 6;  // OK: x is mutable
    println!("x = {}", x);
}

Constants


const MAX_POINTS: u32 = 100_000;
const PI: f64 = 3.14159;

fn main() {
    println!("Max points: {}", MAX_POINTS);
}

Rules:

  • Must have type annotation
  • Use SCREAMING_SNAKE_CASE
  • Can be declared in any scope
  • Always immutable

Shadowing


fn main() {
    let x = 5;
    let x = x + 1;      // Shadows previous x
    let x = x * 2;      // Shadows again

    println!("x = {}", x);  // Prints: x = 12
}

Shadowing allows type changes:


let spaces = "   ";        // &str
let spaces = spaces.len(); // usize

Section 2

Control Flow

if Expressions


fn main() {
    let number = 7;

    if number < 5 {
        println!("less than 5");
    } else if number < 10 {
        println!("between 5 and 10");
    } else {
        println!("10 or greater");
    }
}
Note: Condition must be bool

if as an Expression


fn main() {
    let condition = true;
    let number = if condition { 5 } else { 6 };

    println!("number = {}", number);
}

Both branches must return the same type

loop


fn main() {
    let mut counter = 0;

    let result = loop {
        counter += 1;
        if counter == 10 {
            break counter * 2;  // Returns 20
        }
    };

    println!("Result: {}", result);
}

while Loop


fn main() {
    let mut number = 3;

    while number != 0 {
        println!("{}!", number);
        number -= 1;
    }

    println!("LIFTOFF!");
}

for Loop


fn main() {
    let a = [10, 20, 30, 40, 50];

    for element in a {
        println!("Value: {}", element);
    }

    // With range
    for i in 1..4 {
        println!("{}", i);  // 1, 2, 3
    }

    // Inclusive range
    for i in 1..=4 {
        println!("{}", i);  // 1, 2, 3, 4
    }
}

break and continue


fn main() {
    for i in 1..10 {
        if i == 3 {
            continue;  // Skip this iteration
        }
        if i == 7 {
            break;     // Exit the loop
        }
        println!("{}", i);
    }
}
// Output: 1, 2, 4, 5, 6

Section 3

Functions

Function Syntax


fn main() {
    greet();
    print_value(42);
    let sum = add(3, 4);
    println!("Sum: {}", sum);
}

fn greet() {
    println!("Hello!");
}

fn print_value(x: i32) {
    println!("Value: {}", x);
}

fn add(a: i32, b: i32) -> i32 {
    a + b  // No semicolon = return value
}

Statements vs Expressions


fn main() {
    // Statement - ends with semicolon, no value
    let x = 5;

    // Expression - evaluates to a value
    let y = {
        let a = 3;
        a + 1  // No semicolon - returns 4
    };

    println!("y = {}", y);
}

Return Values


fn five() -> i32 {
    5  // Implicit return
}

fn absolute(x: i32) -> i32 {
    if x < 0 {
        return -x;  // Early return
    }
    x
}

Section 4

Collections

Vec<T> - Dynamic Array


fn main() {
    let mut v: Vec = Vec::new();
    v.push(1);
    v.push(2);
    v.push(3);

    // Or use vec! macro
    let v2 = vec![1, 2, 3];

    println!("{:?}", v);
}

Accessing Vector Elements


fn main() {
    let v = vec![1, 2, 3, 4, 5];

    let third = v[2];           // Panics if out of bounds
    let third = v.get(2);       // Returns Option<&T>

    match v.get(2) {
        Some(value) => println!("Third: {}", value),
        None => println!("No third element"),
    }
}

Iterating Over Vectors


fn main() {
    let v = vec![10, 20, 30];

    for value in &v {
        println!("{}", value);
    }

    // Mutable iteration
    let mut v2 = vec![1, 2, 3];
    for value in &mut v2 {
        *value *= 2;
    }
}

String


fn main() {
    let mut s = String::new();
    s.push_str("hello");
    s.push(' ');
    s.push_str("world");

    // From literal
    let s2 = String::from("hello");
    let s3 = "hello".to_string();

    println!("{}", s);
}

String Operations


fn main() {
    let s1 = String::from("Hello, ");
    let s2 = String::from("world!");
    let s3 = s1 + &s2;  // s1 is moved

    // Or use format!
    let s4 = String::from("Hello");
    let s5 = String::from("world");
    let s6 = format!("{}, {}!", s4, s5);
}

HashMap


use std::collections::HashMap;

fn main() {
    let mut scores = HashMap::new();
    scores.insert(String::from("Blue"), 10);
    scores.insert(String::from("Red"), 50);

    let team = String::from("Blue");
    let score = scores.get(&team);  // Option<&V>

    for (key, value) in &scores {
        println!("{}: {}", key, value);
    }
}
